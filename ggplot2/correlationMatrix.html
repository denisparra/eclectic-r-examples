# Redefinition of function to fit different text size
corrplotd.mixed <- function(corr, lower = "number", upper = "circle", tl.pos = c("d", 
    "lt", "n"), diag = c("n", "l", "u"), bg = "white", addgrid.col = "gray", 
    corr.cex = 1, ...) {
    diag <- match.arg(diag)
    tl.pos <- match.arg(tl.pos)
    n <- nrow(corr)
    corrplot(corr, type = "upper", method = upper, diag = TRUE, tl.pos = tl.pos, 
        ...)
    corrplotd(corr, add = TRUE, type = "lower", method = lower, diag = (diag == 
        "l"), tl.pos = "n", cl.pos = "n", corr.cex = corr.cex, ...)
    if (diag == "n" & tl.pos != "d") {
        symbols(1:n, n:1, add = TRUE, bg = bg, fg = addgrid.col, inches = FALSE, 
            squares = rep(1, n))
    }
}
